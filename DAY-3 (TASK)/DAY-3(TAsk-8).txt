public class InvertedTriangle {
    public static void main(String[] args) {
        int n = 5; 
        for (int i = n; i >= i; i--) {      
            for (int j = 1; j <= i; j++) {   
                System.out.print("* ");
            }
            System.out.println(); 
        }
    }
}
